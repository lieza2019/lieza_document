\begin{Definition}{論理的同値性(INTUITIONISTIC EQUIVALENCE OVER CANONICAL
    TERMS)}
\label{equality_on_CANONICALS}
  項$t$の論理的同値項(INTUITIONISTIC EQUIVALENTS OVER CANONICAL TERMS) $t'$
  を以下のように定義する.
  $t$が$t'$と論理的同値であることを$t \simeq t'$と表記する.
  \begin{displaymath}
    \begin{array}{c}
      
      \begin{array}{rl}
        \begin{array}{rl}
          \dfrac{t \equiv t'}{
            t \simeq t'
          }  &  (\text{\it Identity})
        \end{array}
        & \begin{array}{rl}
            \dfrac{t_1 \simeq t_2 \quad t_2 \simeq t_3}{
              t_1 \simeq t_3
            }  &  (\text{\it Transitivity})
          \end{array}
      \end{array}  \\
      \\

      \begin{array}{rl}
        \dfrac{
          t \simeq t'\
           \text{\it s.t the last applied rule for}\ t \simeq t'\
            \text{\it isn't}\ (\text{\it Symmetry})
        }{
          t' \simeq t
        }  &  (\text{\it Symmetry})
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{t_1 \simeq t'_1 \quad t_2 \simeq t'_2}{
            t_1 \!\wedge\! t_2 \simeq t'_1 \!\wedge\! t'_2
          }  &  (\text{\it Transparency-Cas})
        \end{array}
        
        & \begin{array}{rl}
            \dfrac{t_1 \simeq t'_1 \quad t_2 \simeq t'_2}{
              t_1 \!\vee\! t_2 \simeq t'_1 \!\vee\! t'_2
            }  &  (\text{\it Transparency-Par})
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \dfrac{ t \simeq t_1 \wedge (t_2 \wedge t_3) \quad
          t' \simeq (t_1 \wedge t_2) \wedge t_3 }{
          t \simeq t'
        }  &  (\text{\it Associativity-Cas})
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \dfrac{ t \simeq t_1 \vee (t_2 \vee t_3) \quad
          t' \simeq (t_1 \vee t_2) \wedge t_3 }{
          t \simeq t'
        }  &  (\text{\it Associativity-Par})
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ t \simeq t' }{ t \simeq \{t'\} }
          & (\text{\it Associativity-Cat0}_1)
        \end{array}
        & \begin{array}[c]{l}
            \begin{array}{rl}
              \dfrac{
                t \simeq t_1 \wedge \{t_2 \cdots t_n \}
              }{
                t \simeq \{t_1, t_2 \cdots t_n \}
              }  &  (\text{\it AssociativityH-Cat0}_\infty)
            \end{array}  \\
            \\
            \begin{array}{rl}
              \dfrac{
                t \simeq \{t_1 \cdots t_n \} \wedge t_{n+1}
              }{
                t \simeq \{t_1 \cdots t_n, t_{n+1} \}
              }  &  (\text{\it AssociativityT-Cat0}_\infty)
            \end{array}
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ t \simeq t' }{ t \simeq [t'] }
          & (\text{\it Associativity-Cat1}_1)
        \end{array}
        & \begin{array}[c]{l}
            \begin{array}{rl}
              \dfrac{
                t \simeq t_1 \wedge [t_2 \cdots t_n ]
              }{
                t \simeq [t_1, t_2 \cdots t_n ]
              }  &  (\text{\it AssociativityH-Cat1}_\infty)
            \end{array}  \\
            \\
            \begin{array}{rl}
              \dfrac{
                t \simeq [t_1 \cdots t_n ] \wedge t_{n+1}
              }{
                t \simeq [t_1 \cdots t_n, t_{n+1} ]
              }  &  (\text{\it AssociativityT-Cat1}_\infty)
            \end{array}
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ t \simeq t' }{ t \simeq \langle t' \rangle }
          & (\text{\it Associativity-Dup}_1)
        \end{array}
        & \begin{array}[c]{l}
            \begin{array}{rl}
              \dfrac{
                t \simeq t_1 \vee \langle t_2 \cdots t_n \rangle
              }{
                t \simeq \langle t_1, t_2 \cdots t_n \rangle
              }  &  (\text{\it AssociativityH-Dup}_\infty)
            \end{array}  \\
            \\
            \begin{array}{rl}
              \dfrac{
                t \simeq \langle t_1 \cdots t_n \rangle \vee t_{n+1}
              }{
                t \simeq \langle t_1 \cdots t_n, t_{n+1} \rangle
              }  &  (\text{\it AssociativityT-Dup}_\infty)
            \end{array}
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ \{\,\} \wedge t \simeq t' }{
            t \simeq t'
          }  &  (\text{\it RevealH-Cas})
        \end{array}
        & \begin{array}{rl}
            \dfrac{ t \wedge \{\,\} \simeq t' }{
              t \simeq t'
            }  &  (\text{\it RevealT-Cas})
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ \{\,\} \vee t \simeq t' }{
            t \simeq t'
          }  &  (\text{\it RevealH-Par})
        \end{array}
        & \begin{array}{rl}
            \dfrac{ t \vee \{\,\} \simeq t' }{
              t \simeq t'
            }  &  (\text{\it RevealT-Par})
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ t \simeq t' }{
            \{\,\} \wedge t \simeq t'
          }  &  (\text{\it PhonyH-Cas})
        \end{array}
        & \begin{array}{rl}
            \dfrac{ t \simeq t' }{
              t \wedge \{\,\} \simeq t'
            }  &  (\text{\it PhonyT-Cas})
          \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{ll}
        \begin{array}{rl}
          \dfrac{ t \simeq t' }{
            \{\,\} \vee t \simeq t'
          }  &  (\text{\it PhonyH-Par})
        \end{array}
        & \begin{array}{rl}
            \dfrac{ t \simeq t' }{
              t \vee \{\,\} \simeq t'
            }  &  (\text{\it PhonyT-Par})
          \end{array}
      \end{array}
      
    \end{array}
  \end{displaymath}
  
  \begin{displaymath}
    \begin{array}{c}
      
      \begin{array}{rl}
        \dfrac{
          t \simeq t'
           \text{\it s.t the last applied rule for}\ t \simeq t'\
            \text{\it isn't}\ (\text{\it Symmetry})
        }{
          {t}^? \simeq t'
        }  &  (\text{\it Optional-L})
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \dfrac{
          t \simeq t'
           \text{\it s.t the last applied rule for}\ t \simeq t'\
            \text{\it isn't}\ (\text{\it Symmetry})
        }{
          t \simeq {t'}^?
        }  &  (\text{\it Optional-R})
      \end{array}  \\
      \\

      \begin{array}{rl}
        \dfrac{
          t \simeq t'
          \text{\it s.t the last applied rule for}\ t \simeq t'\
           \text{\it isn't}\ (\text{\it Symmetry})
        }{
          t^\leftarrow \simeq t'
        }  &  (\text{\it AltL-L})
      \end{array}  \\
      \\

      \begin{array}{rl}
        \dfrac{
          t \simeq t'
          \text{\it s.t the last applied rule for}\ t \simeq t'\
          \text{\it isn't}\ (\text{\it Symmetry})
        }{
          t \simeq {t'}^\leftarrow
        }  &  (\text{\it AltL-R})
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \dfrac{
          t \simeq t'
          \text{\it s.t the last applied rule for}\ t \simeq t'\
           \text{\it isn't}\ (\text{\it Symmetry})
        }{
          t^\rightarrow \simeq t'
        }  &  (\text{\it AltR-L})
      \end{array}  \\
      \\

      \begin{array}{rl}
        \dfrac{
          t \simeq t'
          \text{\it s.t the last applied rule for}\ t \simeq t'\
          \text{\it isn't}\ (\text{\it Symmetry})
        }{
          t \simeq {t'}^\rightarrow
        }  &  (\text{\it AltR-R})
      \end{array}  \\
      
    \end{array}
  \end{displaymath}
\end{Definition}

\begin{Definition}{拡張された型束縛(BINDING PENTAGRAM)}
  拡張された型束縛$\Delta$を, 以下のように定義する.
  \begin{displaymath}
    \Delta\ \text{\it s.t.}\ \Delta \!\vdash\! t
     \triangleright \!\mathcal{R} \;\stackrel{\mathrm{def}}{:=}
      \!\!\begin{array}[t]{l}
        \bigl\{(t, t_{\text{\it equ}}, \mathcal{R},
         \mathcal{S}_{\mathpzc{fin}}, \Delta') \bigr\} \cup \Delta'\
          ,\text{\it where}\ \begin{array}[t]{l}
            t \simeq t_{\text{\it equ}}  \\
            \mathcal{S}_{\mathpzc{fin}}\in \{\text{\it gnd},
             \text{\it def}, \wedge, \vee, \text{\it nil}, \text{\it sol},
              \infty, \text{\it L}, \text{\it R}\}
          \end{array}
      \end{array}  \\
  \end{displaymath}
\end{Definition}

\begin{Definition}
  拡張された型束縛を、以下のような対応
  $\mathbb{T} \mapsto \bigl\{(t, t_{\text{\it equ}}, \mathcal{R},
  \mathcal{S}_{\mathpzc{fin}}, \Delta') \bigr\}$と定義する.
  \begin{displaymath}
    \Delta(t) \;\stackrel{\mathrm{def}}{:=} \,\left\{\begin{array}{l}
      \bigl\{(t', t'_{\text{\it equ}}, \mathcal{R},
       \mathcal{S}_{\mathpzc{fin}}, \Delta') \bigr\} \quad \text{\it if}\
        t' = t  \\
      \Delta'(t) \quad \text{\it if}\ t' \neq t,\ \Delta' \neq \phi  \\
      \phi \quad \text{\it if}\ t' \neq t,\ \Delta' = \phi  \\
    \end{array}\right. \begin{array}{r}
      ,\text{\it where}\
       \Delta = \bigl\{(t, t_{\text{\it equ}}, \mathcal{R},
        \mathcal{S}_{\mathpzc{fin}}, \Delta') \bigr\} \cup \Delta'  \\
      \text{\it for some}\ (t', t'_{\text{\it equ}},
       \mathcal{R}, \mathcal{S}_{\mathpzc{fin}}, \Delta') \in \Delta.
    \end{array}
  \end{displaymath}
\end{Definition}


\begin{Definition}{拡張された正準表現項に関する型照合(汎照合:POLYMORPHIC

\begin{Lemma}{soundness on extended matching}
  \label{soundness_on_extended_matching}
  \[ \text{\it if}\ \Gamma \vdash t : \mathcal{R},\
      \text{\it then}\ \Delta \vdash t \triangleright \mathcal{R}\
       \text{\it s.t.}\ \mathpzc{Dom}(\Delta) = \mathpzc{Dom}(\Gamma)
  \]
  \begin{Proof}
    Proof is structual induction on the derivation of
    $\Gamma \vdash t : \mathcal{R}$ and term $t$.

    Case (T-Atom0-canon): Proof for this case is trivial, which is directly
    from both def. of (T-Atom0-canon) and (T-Atom0-xtend), as follow.
    Here we can assume that $t$ satisfies $(t, \mathcal{R}_0) \in
    \Gamma_\text{\it gnd}$ from def. of (T-Atom0-canon) , then we can
    derive the equation of
    $\bigl\{(t, t, \mathcal{R}_0, \text{\it gnd}, \phi) \bigr\} \vdash
    t \triangleright \mathcal{R_0}$ from (T-Atom0-xtend), because of
    $t \cong t$ s.t. $(t, \mathcal{R}_0) \in \Gamma_{gnd}$ from (Identity),
    which satisfies the condition of
    $\mathpzc{Dom}\Bigl(\bigl\{(t, \mathcal{R}_0 ) \bigr\} \Bigr) =
    \{t\} = \bigl(\{t\} \cup \phi \bigr) =
    \mathpzc{Dom}\Bigl(\bigl\{(t, t, \mathcal{R}_0, \text{\it gnd}, \phi)
    \bigr\} \Bigr)$ as desired.
    
    Case (T-Atom1-canon): Proof is same scheme as the case of
    (T-Atom0-canon).
    
    Case (T-Cas-canon): From I.H. we can assume that
    $\Delta_1 \vdash t_1 \triangleright \mathcal{R}_1$,
    $\Delta_2 \vdash t_2 \triangleright \mathcal{R}_2$ s.t.
    $\mathpzc{Dom}(\Delta_1) = \mathpzc{Dom}(\Gamma_1)$ and
    $\mathpzc{Dom}(\Delta_2) = \mathpzc{Dom}(\Gamma_2)$ respectively.
    That implies the fact of
    $\mathpzc{Dom}(\Delta_1) \cap \mathpzc{Dom}(\Delta_2) = \phi$
    because of
    $\mathpzc{Dom}(\Gamma_1) \cap \mathpzc{Dom}(\Gamma_2) = \phi$.
    Similarly we still find the fact that $t_1 \wedge t_2$ isn't belonging
    to $\bigl(\mathpzc{Dom}(\Gamma_1) \cup \mathpzc{Dom}(\Gamma_1) \bigr)
    = \bigl(\mathpzc{Dom}(\Delta_1) \cup \mathpzc{Dom}(\Delta_1) \bigr)$.    
    Then we derive the equivalency that
    $\Bigl\{\bigl(t_1 \wedge t_2, t_1 \wedge t_2,
    \mathcal{R}_1 \wedge \mathcal{R}_2, (\Delta_1 \cup \Delta_2) \bigr)
    \Bigr\} \cup (\Delta_1 \cup \Delta_2) \vdash t_1 \wedge t_2
    \triangleright \mathcal{R}_1 \wedge \mathcal{R}_2$ from (T-Cas-xtend),
    with desired condition as
    $\mathpzc{Dom}\biggl(\Bigl\{\bigl(t_1 \wedge t_2, t_1 \wedge t_2,
    \mathcal{R}_1 \wedge \mathcal{R}_2, (\Delta_1 \cup \Delta_2) \bigr)
    \Bigr\} \cup (\Delta_1 \cup \Delta_2) \biggr) =
    \bigl(\{t_1 \wedge t_2\} \cup \mathpzc{Dom}(\Delta_1 \cup \Delta_2)
    \bigr) =
    \Bigl(\{t_1 \wedge t_2\} \cup \bigl(\mathpzc{Dom}(\Delta_1) \cup
    \mathpzc{Dom}(\Delta_2) \bigr) \Bigr) =
    \Bigl(\{t_1 \wedge t_2\} \cup \bigl(\mathpzc{Dom}(\Gamma_1) \cup
    \mathpzc{Dom}(\Gamma_2) \bigr) \Bigr) =
    \bigl(\{t_1 \wedge t_2\} \cup \mathpzc{Dom}(\Gamma_1 \cup \Gamma_2)
    \bigr) = \mathpzc{Dom}(\Gamma)$.
    
    Case (T-Par-canon): Proof is same scheme as the case of
    (T-Cat-canon),
    
    Case (T-Cat0-canon-nil): It's trivial, proof for this case is directly
    from def, of (T-Cat0-canon-nil) and (T-Cat0-xtend-nil),

    $\Delta_j \vdash t_j : \mathcal{R}_j$ s.t.
    $\mathpzc{Dom}(\Gamma_j) = \mathpzc{Dom}(\Delta_j)$ for each,

    $j = 1$ and $j \ge 2$.
    $(j = 1)$: For first case, we can derive the term $\{t_1\}$ s.t.

    $\{t_1\} \not\in \mathpzc{Dom}(\Gamma_1) = \mathpzc{Dom}(\Delta_1)$.

    \triangleright {\mathcal{R}_1}^*$ from (T-Cat0-xtend-sol),

    \bigl(\{t_1\} \cup \mathpzc{Dom}(\Gamma_1) \bigr) =
    \mathpzc{Dom}(\Gamma)$, where we let
    $\Gamma = \Bigl\{\bigl(\{t_1\}, {\mathcal{R}_1}^* \bigr) \Bigr\} \cup
    \Gamma_1$.
    $(j \ge 2)$: Here we think of the case of $j = 2$ that

    And next, we think the case of $j > 2$. Here we suppose the situation of

    $\Gamma_2 \vdash t_2 : {\mathcal{R}_1}^* \cdots

    $\{t_2 \cdots t_n\} \cup \bigcup_{j=2}^n\Gamma_j \vdash
    \{t_2 \cdots t_n\}: {\mathcal{R}_1}^*$ from (T-Cat0-canon) with same
    size of derivation for
    $\Bigl\{\bigl(\{t_1 \cdots t_n\}, {\mathcal{R}_1}^* \bigr) \Bigr\} \cup
    \bigcup_{j=1}^n\Gamma_j \vdash \{t_1 \cdots t_n\}: {\mathcal{R}_1}^*$.

    $t_1 \cdots t_n$ in $\{t_1 \cdots t_n\}$. We let n as number of subterms

    Then we can derive

    \Gamma_j) = \mathpzc{Dom}(\Gamma)$ as desired.
    
    Case (T-Cat1-canon): Proof is same scheme to the case of

\begin{Lemma}
  \label{consistency_on_domain_independency}
  \[ \begin{array}{r}
       \mathpzc{Dom}(\Delta_1) \cap \mathpzc{Dom}(\Delta_2) = \phi\
        \text{\it where}\ \Delta_1 \vdash t_1 \triangleright \mathcal{R}_1\
         \text{\it and}\ \Delta_2 \vdash t_2 \triangleright \mathcal{R}_2,\
          \text{\it then}  \\
       \mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr) \cap
        \mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr) = \phi
     \end{array}
  \]
  \begin{Proof}
    From def. of $\mathpzc{sub}$ over $\mathbb{T}$, any term in
    $\mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr)$
    consists of the grand terms belonging to
    $\mathpzc{gnd}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr)$, which
    holds on any terms in 
    $\mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_2(t_2) \bigr) \Bigr)$ e.g.
    all of grand terms occuring in
    $\mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_2(t_2) \bigr) \Bigr)$ come
    from $\mathpzc{gnd}\Bigl(\mathcal{M}\bigl(\Delta_2(t_2) \bigr) \Bigr)$.
    We mention the facts that
    $\mathpzc{gnd}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr) =
    {\mathpzc{Gnd}_\Delta}(\Delta_1) \subseteq \mathpzc{Dom}(\Delta_1)$ and
    $\mathpzc{gnd}\Bigl(\mathcal{M}\bigl(\Delta_1(t_2) \bigr) \Bigr) =
    {\mathpzc{Gnd}_\Delta}(\Delta_2) \subseteq \mathpzc{Dom}(\Delta_2)$ from
    \ref{consistency_on_grand_terms} and def of $\mathpzc{Dom}$, then we
    find that
    $\mathpzc{gnd}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr) \cap
    \mathpzc{gnd}\Bigl(\mathcal{M}\bigl(\Delta_2(t_2) \bigr) \Bigr) = \phi$
    because of $\mathpzc{Dom}(\Delta_1) \cap \mathpzc{Dom}(\Delta_2) =
    \phi$. Hence we can conclude that
    $\mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_1(t_1) \bigr) \Bigr) \cap
    \mathpzc{sub}\Bigl(\mathcal{M}\bigl(\Delta_2(t_2) \bigr) \Bigr) = \phi$.
    $\Box$
  \end{Proof}
\end{Lemma}

    $t_{\text{\it equ}_{t_h}} \wedge t'_{\text{\it equ}_{t_t}} \not\in

    $\mathcal{M}_{\{\}}\bigl(

    $\{t_2, \cdots t_n\} = \Bigl\{\mathcal{M}\bigl(\Delta'_t(

    \Bigl\{\mathcal{M}\bigl(\Delta_{t_h}(t_{\text{\it equ}_{t_h}}) \bigr),
