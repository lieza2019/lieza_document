\begin{Definition}{実装を鑑みた論理的同値性(INTUITIONISTIC EQUIVALENCE
    FOR IMPLEMENTATION)}
  \label{equality_on_implementation}
  項$t$の実装を鑑みた論理的同値項(INTUITIONISTIC EQUIVALENTS FOR
  IMPLEMENTATION) $t'$を以下のように定義する.
  $t$が$t'$と論理的同値であることを$t \approxeq t'$と表記する.
  \begin{displaymath}
    t \approxeq t' \;\stackrel{\mathrm{def}}{:=}\; t' \in \mathcal{E}(t)\
     \text{\it where}
  \end{displaymath}
  \begin{displaymath}
    \begin{array}{c}

      \begin{array}{rl}
        \begin{array}{rl}
          t \in \mathcal{E}(t)  &  (\text{\it Identity})
        \end{array}
        
        &  \begin{array}{rl}
             t \in \mathcal{E}(t')\ \text{\it s.t.}\
              t' \in \mathcal{E}(t)
             &  (\text{\it Symmetry})
           \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        t'_2 \in \mathcal{E}(t_1)\ \text{\it s.t.}\
         t'_2 \in \mathcal{E}(t_2)\ \text{\it forall}\
          t_2 \in \mathcal{E}(t_1)
        &  (\text{\it Transitivity})
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        t'_1 \wedge t'_2 \in \mathcal{E}(t_1 \wedge t_2)\
         \text{\it s.t.}\ t'_1 \in \mathcal{E}(t_1),\
          t'_2 \in \mathcal{E}(t_2)
        &  (\text{\it Transparency-Cas})
      \end{array}  \\
           
      \begin{array}{rl}
        t'_1 \vee t'_2 \in \mathcal{E}(t_1 \vee t_2)\
         \text{\it s.t.}\ t'_1 \in \mathcal{E}(t_1),\
          t'_2 \in \mathcal{E}(t_2)
        &  (\text{\it Transparency-Par})
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        t_1 \wedge (t_2 \wedge t_3) \in
         \mathcal{E}\bigl((t_1 \wedge t_2) \wedge t_3 \bigr)
        &  (\text{\it Associativity-Cas-L})
      \end{array}  \\
      
      \begin{array}{rl}
        (t_1 \wedge t_2) \wedge t_3 \in
         \mathcal{E}\bigl(t_1 \wedge (t_2 \wedge t_3) \bigr)
        &  (\text{\it Associativity-Cas-R})
      \end{array}  \\
      \\

      \begin{array}{rl}
        t_1 \vee (t_2 \vee t_3) \in
         \mathcal{E}\bigl((t_1 \vee t_2) \wedge t_3 \bigr)
        &  (\text{\it Associativity-Par-L})
      \end{array}  \\
      
      \begin{array}{rl}
        (t_1 \vee t_2) \vee t_3 \in
         \mathcal{E}\bigl(t_1 \vee (t_2 \wedge t_3) \bigr)
        &  (\text{\it Associativity-Par-R})
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \{t'\} \in \mathcal{E}(t)\ \text{\it s.t.}\ t' \in \mathcal{E}(t)
        & (\text{\it Associativity-Cat0}_1)
      \end{array}  \\

      \begin{array}{rl}
        \{t_1, t_2, \cdots t_n\} \in \mathcal{E}(t)\ \text{\it s.t.}\
         t_1 \wedge \{t_2, \cdots t_n\} \in \mathcal{E}(t)
        &  (\text{\it AssociativityH-Cat0}_\infty)
      \end{array}  \\
      
      \begin{array}{rl}
        \{t_1,\cdots t_n, t_{n+1}\} \in \mathcal{E}(t)\ \text{\it s.t.}\
         \{t_1, \cdots t_n\} \wedge t_{n+1} \in \mathcal{E}(t)
        &  (\text{\it AssociativityT-Cat0}_\infty)
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        [t'] \in \mathcal{E}(t)\ \text{\it s.t.}\ t' \in \mathcal{E}(t)
        &  (\text{\it Associativity-Cat1}_1)
      \end{array}  \\
      
      \begin{array}{rl}
        [t_1, t_2, \cdots t_n]  \in \mathcal{E}(t)\ \text{\it s.t.}\
         t_1 \wedge [t_2, \cdots t_n] \in \mathcal{E}(t)
        &  (\text{\it AssociativityH-Cat1}_\infty)
      \end{array}  \\
      
      \begin{array}{rl}
        [t_1,\cdots t_n, t_{n+1}] \in \mathcal{E}(t)\ \text{\it s.t.}\
         [t_1, \cdots t_n] \wedge t_{n+1} \in \mathcal{E}(t)
        &  (\text{\it AssociativityT-Cat1}_\infty)
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \langle t'\rangle \in \mathcal{E}(t)\ \text{\it s.t.}\
         t' \in \mathcal{E}(t)
        & (\text{\it Associativity-Dup}_1)
      \end{array}  \\
      
      \begin{array}{rl}
        \langle t_1, t_2, \cdots t_n\rangle  \in \mathcal{E}(t)\
         \text{\it s.t.}\
          t_1 \vee \langle t_2, \cdots t_n\rangle \in \mathcal{E}(t)
        &  (\text{\it AssociativityH-Dup}_\infty)
      \end{array}  \\
      
      \begin{array}{rl}
        \langle t_1,\cdots t_n, t_{n+1}\rangle \in \mathcal{E}(t)\
         \text{\it s.t.}\
         \langle t_1, \cdots t_n\rangle \vee t_{n+1} \in \mathcal{E}(t)
        &  (\text{\it AssociativityT-Dup}_\infty)
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \begin{array}{rl}
          t' \in \mathcal{E}(t)\ \text{\it s.t.}\
           \{\} \wedge t' \in \mathcal{E}(t)
          &  (\text{\it RevealH-Cas})
        \end{array}
        
        &  \begin{array}{rl}
             t' \in \mathcal{E}(t)\ \text{\it s.t.}\
              t' \wedge \{\} \in \mathcal{E}(t)
             &  (\text{\it RevealT-Cas})
           \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        \begin{array}{rl}
          t' \in \mathcal{E}(t)\ \text{\it s.t.}\
           \{\} \vee t' \in \mathcal{E}(t)
          &  (\text{\it RevealH-Par})
        \end{array}
        
        &  \begin{array}{rl}
             t' \in \mathcal{E}(t)\ \text{\it s.t.}\
              t' \vee \{\} \in \mathcal{E}(t)
             &  (\text{\it RevealT-Par})
           \end{array}
      \end{array}  \\
      \\

      \begin{array}{rl}
        \begin{array}{rl}
          \{\} \wedge t' \in \mathcal{E}(t)\ \text{\it s.t.}\
          t' \in \mathcal{E}(t)
          &  (\text{\it PhonyH-Cas})
        \end{array}
        
        &  \begin{array}{rl}
             t' \wedge \{\} \in \mathcal{E}(t)\ \text{\it s.t.}\
              t' \in \mathcal{E}(t)
             &  (\text{\it PhonyT-Cas})
           \end{array}
      \end{array}  \\
      \\

      \begin{array}{rl}
        \begin{array}{rl}
          \{\} \vee t' \in \mathcal{E}(t)\ \text{\it s.t.}\
          t' \in \mathcal{E}(t)
          &  (\text{\it PhonyH-Par})
        \end{array}
        
        &  \begin{array}{rl}
             t' \vee \{\} \in \mathcal{E}(t)\ \text{\it s.t.}\
              t' \in \mathcal{E}(t)
             &  (\text{\it PhonyT-Par})
           \end{array}
      \end{array}  \\
      \\
      
      \begin{array}{rl}
        {t'}^? \in \mathcal{E}(t)\ \text{\it s.t.}\
         t' \in \mathcal{E}(t)
        &  (\text{\it Optional})
      \end{array}  \\
      \\

      \begin{array}{rl}
        \begin{array}{rl}
          {t'}^\leftarrow \in \mathcal{E}(t)\ \text{\it s.t.}\
           t' \in \mathcal{E}(t)
          &  (\text{\it AltL})
        \end{array}
        
        &  \begin{array}{rl}
             {t'}^\rightarrow \in \mathcal{E}(t)\ \text{\it s.t.}\
              t' \in \mathcal{E}(t)
             &  (\text{\it AltR})
           \end{array}
      \end{array}
      
    \end{array}
  \end{displaymath}
\end{Definition}
